264 bool
265 RandomVariableInputWidget::inputFromJSON(QJsonObject &rvObject)
266 {
267  bool result = true;
268 
269  // clean out current list
270  this->clear();
271 
272  //
273  // go get randomvariables array from the JSON object
274  // for each object in array:
275  // 1)get it'is type,
276  // 2)instantiate one
277  // 4) get it to input itself
278  // 5) finally add it to layout
279  //
280 
281  // get array
282  if (rvObject.contains("randomVariables"))
283  if (rvObject["randomVariables"].isArray()) {
284 
285  QJsonArray rvArray = rvObject["randomVariables"].toArray();
286 
287  // foreach object in array
288  foreach (const QJsonValue &rvValue, rvArray) {
289 
290  QJsonObject rvObject = rvValue.toObject();
291 
292  if (rvObject.contains("variableClass")) {
293  QJsonValue typeRV = rvObject["variableClass"];
294  RandomVariable *theRV = 0;
295  QString classType = typeRV.toString();
296  theRV = new RandomVariable(classType);
297  connect(theRV,SIGNAL(sendErrorMessage(QString)),this,SLOT(errorMessage(QString)));
298 
299  if (theRV->inputFromJSON(rvObject)) { // this method is where type is set
300  theRandomVariables.append(theRV);
301  rvLayout->insertWidget(rvLayout->count()-1, theRV);
302  } else {
303  result = false;
304  }
305  } else {
306  result = false;
307  }
308  }
309  }
310  return result;
311 }
